% ---------------------------------------------------------------
% Chapter / section formatting
%
% NOTE: \usepackage{parskip} sets \parskip ~ 6-7 pt globally.
% In [display] mode titlesec adds \parskip to the sep argument,
% so sep must be reduced by that amount to keep the visual gap
% between the "Chapter N" label and the title at ~10 pt.
% Similarly, \titlespacing after-skip is set to 0 pt so that the
% space below the titlerule is supplied only by \parskip (~7 pt)
% plus the tcolorbox before skip (10 pt), giving ~17 pt total —
% enough breathing room without the large gap that arises when
% the after-skip, \parskip, and before-skip all stack.
% ---------------------------------------------------------------
\titleformat{\chapter}[display]
  {\LARGE\bfseries\color{isublue}}
  {\large\color{accent}Chapter \thechapter}{3pt}
  {\LARGE\bfseries\color{isublue}}
  [\vspace{4pt}\color{accent}\titlerule]

\titleformat{\section}
  {\Large\bfseries\color{isublue}}
  {\thesection}{1em}{}
  [\color{accent}\titlerule]

\titleformat{\subsection}
  {\large\bfseries\color{accent}}
  {\thesubsection}{1em}{}

\titlespacing*{\chapter}{0pt}{0pt}{0pt}
\titlespacing*{\section}{0pt}{18pt}{8pt}
\titlespacing*{\subsection}{0pt}{12pt}{4pt}
